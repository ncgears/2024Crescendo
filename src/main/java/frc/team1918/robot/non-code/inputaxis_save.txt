## create an input axis wrapper for analog joysticks, then use them like

private InputAxis m_fwdYAxis = new InputAxis('Forward',dj::getLeftY)
    .withDeadband(0.10)
    .withSquaring(true)
    .withInvert(true);
private InputAxis m_fwdXAxis = new InputAxis('Strafe',dj::getLeftX)
    .withDeadband(0.10)
    .withSquaring(true)
    .withInvert(true);
private InputAxis m_fwdYAxis = new InputAxis('Rotate',dj::getRightX)
    .withDeadband(0.10);
